MyNeuralNet(
  (layer1): Linear(in_features=5, out_features=10, bias=True)
  (layer2): Linear(in_features=10, out_features=2, bias=True)
)



def forward(self, x):
    layer1 = self.layer1(x);  x = None
    layer2 = self.layer2(layer1);  layer1 = None
    relu = torch.nn.functional.relu(layer2, inplace = False);  layer2 = None
    return relu
